const mongoose = require('mongoose'),
  Student = mongoose.Schema,
  validate = require('mongoose-unique-validator');
 
var studentSchema = new Student({
  personalDetails:{
    fullnamename:{
      type: String,
      required: true,
      lowercase: true,
      index: true
    },
    pob: {
      type: String,
      required: true
    },
    dob:{
      type: Date,
      required: true
    },
    parent:{
      type: String,
      required: true
    },
    address:{
      type: String,
      required: true,
      lowercase: true
    },
    religion:{
      type: String,
      lowercase: true,
    },
    gender:{
      type: String,
      required: true,
      lowercase: true
    },
    image:{
      type: String,
      default: ''
    }
  },
  academicDetails:{
    nisn :{
      type: Number,
      required: [true, 'can\'t be empty'],
      minlength: 10
    },
    school:{
      type: Student.Types.ObjectId,
      ref:'school'
    },
    kelas:{
      type: Student.Types.ObjectId,
      ref: 'classroom'
    },
    isGraduated:{
      type: Boolean,
      default: false
    },
    nilai:[{
      type: Student.Types.ObjectId,
      ref: 'subjects'
    }],
    presensi:[{
      type: Student.Types.ObjectId,
      ref: 'presensi'
    }]
  },
  account:{
    email: {
      type: String,
      lowercase: true,
      required: [true,'can\'t be blank'],
      unique: true,
      index: true,
      match: [/\S+@\S+.\S+/, 'wrong format']
    },
    password: {
      type: String,
      required: true,
      index: true,
      minlength: 6
    }
  },
  role: {
    type: String,
    lowercase : true,
    index: true
  }
},
{timestamps:{
  createdAt:'Registered at: ',
  updatedAt:'Data updated at: '
}});
 
studentSchema.plugin(validate);
module.exports = mongoose.model('student',studentSchema);
